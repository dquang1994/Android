Step 1: Drag check box into layout

Step 2: Get reference to check box: (inside onCreate)


import android.widget.CheckBox;

CheckBox cb1 = (CheckBox) findViewById(R.id.checkBox1);

Step 3: setOnCheckedChangeListener on check box: (inside OnCreate)


import android.widget.CompoundButton;

cb1.setOnCheckedChangeListener(new CompoundButton.OnCheckedChangeListener()
{
	public void onCheckedChanged(CompoundButton buttonView, boolean isChecked)
	{
		if(isChecked){
			//Do something when check box is checked
		}else{
			//Do something when check box isn't checked
		}
	}
});
